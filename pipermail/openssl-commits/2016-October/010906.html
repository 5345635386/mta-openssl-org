+static int check_bn(const char *name, const BIGNUM *bn, const char *hexbn)
+{
+    BIGNUM *tmp = NULL;
+    int rv;
+    if (BN_hex2bn(&tmp, hexbn) == 0)
+        return 0;
+    rv = BN_cmp(bn, tmp);
+    if (rv == 0) {
+        printf("%s = ", name);
+        BN_print_fp(stdout, bn);
+        printf("\n");
+        BN_free(tmp);
+        return 1;
+    }
+    printf("Unexpected %s value\n", name);
+    printf("Expecting: ");
+    BN_print_fp(stdout, tmp);
+    printf("\nReceived: ");
+    BN_print_fp(stdout, bn);
+    printf("\n");
+    BN_free(tmp);
+    return 0;
+}
