id-ce-subjectAltName OBJECT IDENTIFIER ::= { id-ce 17 }

Id-on-hardwareModuleName OBJECT IDENTIFIER ::= iso(1)
identified-organization(3) dod(6)

HardwareModuleName ::= SEQUENCE {
