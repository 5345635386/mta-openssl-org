i have this piece of code, but i'm not finding something similar with c++,

import socket
import ssl
import sys
if len(sys.argv) == 3:
   HOST = sys.argv[1]     # IP
   PORT = int(sys.argv[2])  # Port
else:
   print "USAGE: $python client_ssl.py <server_ip> <port>"
   exit(1)
print 'Connecting...'
s = socket.create_connection((HOST, PORT))
s = ssl.wrap_socket(s)
print "Connected!\n"
while True:
   query = raw_input("Query: ")
   if ( query == "quit" ):
      break
   s.send(query)
   data = s.recv(16384)
   print 'Reply:', data
s.close()
