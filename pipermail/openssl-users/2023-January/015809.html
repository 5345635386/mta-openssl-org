  a_XTLS__JFETCH:/* C99 */{
        EVP_MD_CTX *mdcp;

        if((mdcp = EVP_MD_CTX_new()) == NIL)
           *mdp = NIL;

           if(!EVP_DigestInit_ex(mdcp, *mdp, NIL))

     if((cipher = EVP_get_cipherbyname(cp)) != NIL)
